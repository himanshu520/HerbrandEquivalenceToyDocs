173 bool samePartition(std::vector<int> const &first, std::vector<int> const &second) {
174  // map to store equivalent indexes (have same value)
175  // in the first partition
176  std::map<int, std::vector<int>> mp;
177  for(int i = 0; i < (int)first.size(); i++)
178  mp[first[i]].push_back(i);
179 
180  // checking if equivalent indexes in the first
181  // partition are also equivalent in the second.
182  // If not `false` is returned
183  for(auto &el : mp) {
184  int elSetId = second[el.second[0]];
185  for(auto &nel : el.second)
186  if(second[nel] != elSetId) return false;
187  }
188 
189  // the partitions are equivalent, so return `true`
190  return true;
191 }

218 int findSet(std::vector<int> const &partition, Program::ExpressionTy const &exp) {
219  // set identifier corresponding to left and right
220  // subexpressions at the current program point
221  int leftSetId = partition[IndexExp[{'\0', exp.leftOp, NULL_VAL}]];
222  int rightSetId = partition[IndexExp[{'\0', exp.rightOp, NULL_VAL}]];
223 
224  // checking if a set representing the expression
225  // already exists
226  std::tuple<char, int, int> tup = {exp.op, leftSetId, rightSetId};
227  auto it = Parent.find(tup);
228 
229  // if the set already exists return its identifier,
230  // otherwise return new set identifier and update
231  // `Parent` map with this information
232  if(it == Parent.end())
233  return Parent[tup] = SetCnt++;
234  return it->second;
235 }

347 void transferFunction(int cfgIndex) {
348  // current partition vector
349  std::vector<int> &partition = Partitions[cfgIndex];
350 
351  // first copy predecessor partition into current partition
352  partition = Partitions[program.CFG[cfgIndex].predecessors[0]];
353 
354  // if the current partition has any index with value -1, it
355  // means that it represents the TOP element and it has to be
356  // left as such without any modifications
357  if(partition[0] == -1) return;
358 
359  // if the node is the END CFG node, return
360  int instIdx = program.CFG[cfgIndex].instructionIndex;
361  if(instIdx == program.Instructions.size() - 1) return;
362 
363  // `changedExp` is expression (ie. a constant or a variable)
364  // which has been assigned value and `changedToExp` is the
365  // expression which has been assigned to it
366  Program::InstructionTy &inst = program.Instructions[instIdx];
367  Program::ExpressionTy changedExp = {'\0', inst.lValue, NULL_VAL};
368  Program::ExpressionTy changedToExp = inst.rValue;
369 
370  if(changedToExp.op == '#') {
371  // if it is a non-deterministic assignment, then create a
372  // new set identifier
373  partition[IndexExp[changedExp]] = SetCnt++;
374  } else {
375  // assign the `changedExp`, the set identifier of
376  // `changedToExp`
377  partition[IndexExp[changedExp]] = partition[IndexExp[changedToExp]];
378  }
379 
380  // update set identifiers for two length expressions
381  // involving `changedExp`. Here set identifiers of all
382  // two length expressions are assigned values, but only
383  // those involving `changedExp` are updated
384  for(auto &el : IndexExp) {
385  if(el.first.op == '\0') continue;
386  partition[el.second] = findSet(partition, el.first);
387  }
388 }

399 void confluenceFunction(int cfgIndex) {
400  // vector of predecessor CFG node indexes
401  std::vector<int> &predecessors = program.CFG[cfgIndex].predecessors;
402 
403  // if all the predecessors partition represents TOP
404  // element then their confluence is also TOP element.
405  // So, the current partition need not be modified as
406  // by initialisation the current partition is also the
407  // TOP element
408  bool cont = false;
409  for(auto pred : predecessors)
410  cont |= (Partitions[pred][0] != -1);
411  if(not cont) return;
412 
413  // the current partition
414  std::vector<int> &partition = Partitions[cfgIndex];
415 
416  // to check which expressions has already been processed
417  std::vector<bool> accessFlag(IndexExp.size(), false);
418 
419  // process all the expressions one by one
420  for(auto &el : IndexExp) {
421  int elIdx = el.second;
422 
423  // continue if the expression is already processed
424  if(accessFlag[elIdx]) continue;
425 
426  // mark it as processed
427  accessFlag[elIdx] = true;
428 
429  // if the set identifier for the expression in all
430  // the predecessors is same it is assigned the same
431  // set identifier; else a new set identifier is
432  // created and assigned to all the expressions which
433  // belongs to the same equivalence class as the
434  // current expression across all the predecessors and
435  // also mark them as processed
436  int elSetId = -1;
437  bool flag = true;
438  for(auto pred : predecessors) {
439  int elSetIdPred = Partitions[pred][elIdx];
440  if(elSetIdPred == -1) continue;
441  else if(elSetId == -1) elSetId = elSetIdPred;
442  else if(elSetId != elSetIdPred) flag = false;
443  }
444 
445  if(flag) partition[elIdx] = elSetId;
446  else {
447  std::set<int> intersection, elClassPred;
448  for(int i = 0; i < IndexExp.size(); i++)
449  intersection.insert(i);
450 
451  for(auto pred : predecessors) {
452  getClass(Partitions[pred], elIdx, elClassPred);
453  setIntersect<int>(intersection, elClassPred);
454  }
455 
456  int newSetId = SetCnt++;
457  for(auto nel : intersection)
458  accessFlag[nel] = true, partition[nel] = newSetId;
459  }
460  }
461 
462  // now update `Parent` map
463  for(auto &el : IndexExp) {
464  if(el.first.op == '\0') continue;
465 
466  int leftSetId = partition[IndexExp[{'\0', el.first.leftOp, NULL_VAL}]];
467  int rightSetId = partition[IndexExp[{'\0', el.first.rightOp, NULL_VAL}]];
468 
469  Parent[{el.first.op, leftSetId, rightSetId}] = partition[el.second];
470  }
471 }

480 void HerbrandEquivalence() {
481  PRINT_HEADER("Herbrand Equivalence Computation");
482  std::cout << "\n";
483 
484  // assign index to expressions arbitrarily
485  assignIndex();
486 
487  // initialise partition vector with -1 for each program
488  // points and each expression - this stands for TOP
489  // partition at each program point. Note that any element
490  // of partition vector being -1 means that whole vector
491  // holds -1 and represents TOP partition
492  Partitions.assign(program.CFG.size(),
493  std::vector<int>(IndexExp.size(), -1));
494 
495  // initialise starting partition for START node
496  findInitialPartition(Partitions[0]);
497 
498  PRINT_HEADER("Initial Partition");
499  printPartition(Partitions[0]);
500  std::cout << "\n\n\n";
501 
502  bool converged = false;
503  int iterationCtr = 0;
504 
505  // repeat while convergence
506  while(not converged) {
507  PRINT_HEADER("Iteration " + std::to_string(++iterationCtr));
508  converged = true;
509 
510  // for all program points (nodes in CFG) except
511  // START apply transfer/confluence function as
512  // applicable
513  for(int i = 1; i < program.CFG.size(); i++) {
514  std::vector<int> oldPartition = Partitions[i];
515  std::vector<int> &predecessors = program.CFG[i].predecessors;
516  int instIdx = program.CFG[i].instructionIndex;
517 
518  std::cout << '[' << i << "] : ";
519  if(predecessors.size() > 1) {
520  // if CFG node corresponds to a confluence point
521 
522  std::cout << "Confluence of [ ";
523  for(auto el : predecessors)
524  std::cout << el << " ";
525  std::cout << "]\n\t";
526 
527  confluenceFunction(i);
528  } else {
529  // if CFG node corresponds to a transfer point
530 
531  if(instIdx != program.Instructions.size() - 1) {
532  program.cout << "Transfer Point => (" << instIdx << ") "
533  << program.Instructions[instIdx];
534  } else program.cout << "END";
535 
536  program.cout << " [" << predecessors[0] << "]\n\t";
537 
538  transferFunction(i);
539  }
540 
541  printPartition(Partitions[i]);
542  std::cout << "\n\n";
543 
544  // update convergence flag
545  if(not samePartition(oldPartition, Partitions[i]))
546  converged = false;
547  }
548  std::cout << "\n\n";
549  }
550 }
