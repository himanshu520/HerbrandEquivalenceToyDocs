394 void Program::parse(std::string fname) {
395  std::ifstream fin(fname);
396  assert(fin && "Error opening file");
397 
398  // keeps count of the current instruction number. It is
399  // initialised 1 because 0th instruction is the dummy START
400  // instruction
401  int instCnt = 1;
402 
403  // holds a program line
404  std::string buf;
405 
406  // holds jump labels for instructions corresponding to their indexes.
407  // If `jumps` vector is empty for an instruction, it means a default
408  // jump to the instruction corresponding to the next index
409  std::vector<std::vector<std::string>> jumps;
410 
411  // holds the instruction indexes to which the labels refer to
412  std::map<std::string, int> labels;
413 
414  // updating `jumps` and `Program::Instructions` for START instruction
415  InstructionTy dummy = {{false, -1}, {'\0', {false, -1}, {false, -1}}, false, {}};
416  Instructions.emplace_back(dummy);
417  jumps.emplace_back(std::vector<std::string>());
418 
419  // read the program, processing each instruction type and checking
420  // the validity. The jumps can not be resolved at this stage (before
421  // the whole program is processed), as labels can be used before
422  // definitions
423  while(getline(fin, buf)) {
424  // ignores empty lines in the program text
425  if(buf.empty()) continue;
426 
427  // buffer holding current instruction
428  std::stringstream ss(buf);
429 
430  // holds the tokens in the instruction while they are processed
431  // in succession
432  std::string in;
433 
434  // read the first token in the instruction
435  ss >> in;
436 
437  if(in == "GOTO") {
438  while(ss >> in)
439  jumps[instCnt - 1].push_back(in);
440  } else if(in == "LABEL") {
441  // current instruction defines a label
442  while(ss >> in) {
443  assert(labels.find(in) == labels.end() && "Duplicate label found");
444  labels[in] = instCnt;
445  }
446  } else {
447  // current instruction is an assignment instruction
448 
449  int constVal;
450  ValueTy lValue, leftOp, rightOp;
451  char op = '\0';
452 
453  try {
454  // ideally lvalue should be a variable (a string that does
455  // not starts with a digit). So this must throw an exception
456  // and we should reach catch block.
457  constVal = std::stoi(in.c_str());
458 
459  assert(false && "LValue is not a variable");
460  } catch(std::invalid_argument) {
461  // storing lvalue of current instruction
462  lValue = {false, Variables.insert(in).first};
463 
464  // read and ignore the `=` in the instruction buffer
465  in.clear(), ss >> in;
466  assert(in == "=" && "No RValue specified");
467 
468  // read the first operand from the instruction buffer
469  in.clear(), ss >> in;
470  assert(!in.empty() && "No RValue specified");
471 
472  // store left operand of rvalue of current instruction
473  try {
474  constVal = std::stoi(in.c_str());
475  leftOp = {true, Constants.insert(constVal).first};
476  } catch(std::invalid_argument) {
477  if(in == "*") op = '#', leftOp = {false, -1};
478  else leftOp = {false, Variables.insert(in).first};
479  }
480 
481  // read the next token in the instruction buffer (which should
482  // be the operator if any)
483  in.clear(), ss >> in;
484 
485  // if the current instruction is non-deterministic assignment
486  // then it can not have a second operand. Note that if the
487  // rvalue is non-deterministic only then `op` has value `#`.
488  assert((op != '#' || in.empty()) && "Invalid instruction");
489 
490  if(!in.empty()) {
491  // if the current instruction has rvalue, which is two
492  // length expression
493 
494  // determine the operator and check if it is valid.
495  // If the operator is invalid then op will have value
496  // '\0', by which it was initialised
497  if(in.length() == 1) {
498  switch(in[0]) {
499  case '+':
500  case '-':
501  case '*':
502  case '/': op = in[0];
503  }
504  }
505  assert(op != '\0' && "Invalid operand");
506 
507  // read the second operand from the instruction buffer
508  in.clear(), ss >> in;
509  assert(!in.empty() && "Second operand not specified");
510 
511  // store right operand of rvalue of current instruction
512  try {
513  constVal = std::stoi(in.c_str());
514  rightOp = {true, Constants.insert(constVal).first};
515  } catch(std::invalid_argument) {
516  rightOp = {false, Variables.insert(in).first};
517  }
518  } else {
519  // if rvalue is not two length expression, a default value
520  // value for right operand for consistency
521  rightOp = {false, -1};
522  }
523  }
524 
525  // push the current instruction in `Instructions` vector
526  // at index `instCnt`. For now `reachable` and `cfgIndex`
527  // fields are set to default value of false and -1. These
528  // would be set to actual values later.
529  Instructions.push_back({lValue, {op, leftOp, rightOp}, false, -1, {}});
530  jumps.emplace_back(std::vector<std::string>());
531  instCnt++;
532  }
533  }
534 
535  // updating `jumps` and `Program::Instructions` for END instruction
536  Instructions.push_back(dummy);
537  jumps.emplace_back(std::vector<std::string>());
538  instCnt++;
539 
540  // now process the jumps as given by the labels. Reachability
541  // is determined by performing breadth first search (BFS) from
542  // index 0 in `Instructions` (the START instruction).
543  // Instructions for which `jumps` list is empty has by default
544  // jump to the instruction corresponding to the next index.
545  std::queue<int> q;
546  q.push(0), Instructions[0].reachable = true;
547 
548  while(not q.empty()) {
549  int cur = q.front();
550  q.pop();
551 
552  if(not jumps[cur].empty()) {
553  // process the jumps
554  for(auto el : jumps[cur]) {
555  auto it = labels.find(el);
556  assert(it != labels.end() && "Undefined label found");
557 
558  // if the label points to index `instCnt`, it means
559  // the current instruction being processed is the dummy
560  // END instruction and this has to be ignored
561  if(it->second == instCnt) continue;
562 
563  Instructions[it->second].predecessors.insert(cur);
564  if(not Instructions[it->second].reachable) {
565  Instructions[it->second].reachable = true;
566  q.push(it->second);
567  }
568  }
569  } else {
570  // if the index of the next instruction is `instCnt`, it
571  // means the current instruction being processed is the
572  // dummy END instruction and this has to be ignored
573  if(cur + 1 == instCnt) continue;
574 
575  // the successor of current instruction is the next
576  // instruction index
577  Instructions[cur + 1].predecessors.insert(cur);
578  if(not Instructions[cur + 1].reachable) {
579  Instructions[cur + 1].reachable = true;
580  q.push(cur + 1);
581  }
582  }
583  }
584 
585  fin.close();
586 }

639 void Program::createCFG() {
640  // holds the size of the control flow graph
641  int cfgSize = 0;
642 
643  // first set `cfgIndex` for each instruction
644  for(auto &el : Instructions) {
645  // there are no node in CFG corresponding to unreachable
646  // instructions, so we skip them
647  if(not el.reachable) continue;
648 
649  if(el.predecessors.size() > 1) {
650  // if the current instruction is a confluence point.
651  // Note that this condition is not `!= 1`, because it
652  // must also be false for the START instruction (for which
653  // `== 0` is true and hence `!= 1` would then be true)
654 
655  // there would be a node for the instruction as well as
656  // for the confluence point. The confluence point gets
657  // index `cfgSize` and the instruction gets index
658  // `cfgSize + 1`. Only 'cfgSize + 1' is stored, and
659  // the index of confluence point in CFG can be inferred
660  // with its help
661  el.cfgIndex = cfgSize + 1;
662 
663  // update `cfgSize`
664  cfgSize += 2;
665  } else {
666  // there would only be node for the instruction which
667  // gets index `cfgSize` and `cfgSize` is incremented
668  el.cfgIndex = cfgSize++;
669  }
670  }
671 
672  // now create control flow graph, using assigned `cfgIndex`
673  // for each instructions
674  CFG.resize(cfgSize);
675 
676  for(int i = 0; i < Instructions.size(); i++) {
677  InstructionTy &I = Instructions[i];
678 
679  // ignore unreachable instructions
680  if(not I.reachable) continue;
681 
682  int idx = I.cfgIndex;
683  int sz = I.predecessors.size();
684 
685  if(sz > 1) {
686  // if the instruction has more than one predecessors
687 
688  // node for the confluence point
689  int confIdx = idx - 1;
690  CFG[confIdx].instructionIndex = -1;
691  for(auto el : I.predecessors) {
692  int predIdx = Instructions[el].cfgIndex;
693  CFG[confIdx].predecessors.push_back(predIdx);
694  }
695 
696  // node for the instruction
697  CFG[idx].predecessors.push_back(confIdx);
698  CFG[idx].instructionIndex = i;
699  } else {
700  // if the instruction is not the START instruction
701  // update the predecessor for the node corresponding
702  // to the instruction
703  if(sz != 0) {
704  int predIdx = Instructions[*(I.predecessors.begin())].cfgIndex;
705  CFG[idx].predecessors.push_back(predIdx);
706  }
707  CFG[idx].instructionIndex = i;
708  }
709  }
710 }
