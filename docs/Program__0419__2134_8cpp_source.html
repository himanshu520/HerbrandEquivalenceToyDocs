359 void Program::parse(std::string fname) {
360  std::ifstream fin(fname);
361  assert(fin && "Error opening file");
362 
363  // keeps count of the current instruction number. It is
364  // initialised 1 because 0th instruction is the dummy START
365  // instruction
366  int instCnt = 1;
367 
368  // holds a program line
369  std::string buf;
370 
371  // holds jump labels for instructions corresponding to their indexes.
372  // If `jumps` vector is empty for an instruction, it means a default
373  // jump to the instruction corresponding to the next index
374  std::vector<std::vector<std::string>> jumps;
375 
376  // holds the instruction indexes to which the labels refer to
377  std::map<std::string, int> labels;
378 
379  // updating `jumps` and `Program::Instructions` for START instruction
380  InstructionTy dummy = {{false, -1}, {'\0', {false, -1}, {false, -1}}, false, {}};
381  Instructions.emplace_back(dummy);
382  jumps.emplace_back(std::vector<std::string>());
383 
384  // read the program, processing each instruction type and checking
385  // the validity. The jumps can not be resolved at this stage (before
386  // the whole program is processed), as labels can be used before
387  // definitions
388  while(getline(fin, buf)) {
389  // ignores empty lines in the program text
390  if(buf.empty()) continue;
391 
392  // buffer holding current instruction
393  std::stringstream ss(buf);
394 
395  // holds the tokens in the instruction while they are processed
396  // in succession
397  std::string in;
398 
399  // read the first token in the instruction
400  ss >> in;
401 
402  if(in == "GOTO") {
403  while(ss >> in)
404  jumps[instCnt - 1].push_back(in);
405  } else if(in == "LABEL") {
406  // current instruction defines a label
407  while(ss >> in) {
408  assert(labels.find(in) == labels.end() && "Duplicate label found");
409  labels[in] = instCnt;
410  }
411  } else {
412  // current instruction is an assignment instruction
413 
414  int constVal;
415  ValueTy lValue, leftOp, rightOp;
416  char op = '\0';
417 
418  try {
419  // ideally lvalue should be a variable (a string that does
420  // not starts with a digit). So this must throw an exception
421  // and we should reach catch block.
422  constVal = std::stoi(in.c_str());
423 
424  assert(false && "LValue is not a variable");
425  } catch(std::invalid_argument) {
426  // storing lvalue of current instruction
427  lValue = {false, Variables.insert(in).first};
428 
429  // read and ignore the `=` in the instruction buffer
430  in.clear(), ss >> in;
431  assert(in == "=" && "No RValue specified");
432 
433  // read the first operand from the instruction buffer
434  in.clear(), ss >> in;
435  assert(!in.empty() && "No RValue specified");
436 
437  // store left operand of rvalue of current instruction
438  try {
439  constVal = std::stoi(in.c_str());
440  leftOp = {true, Constants.insert(constVal).first};
441  } catch(std::invalid_argument) {
442  if(in == "*") op = '#', leftOp = {false, -1};
443  else leftOp = {false, Variables.insert(in).first};
444  }
445 
446  // read the next token in the instruction buffer (which should
447  // be the operator if any)
448  in.clear(), ss >> in;
449 
450  // if the current instruction is non-deterministic assignment
451  // then it can not have a second operand. Note that if the
452  // rvalue is non-deterministic only then `op` has value `#`.
453  assert((op != '#' || in.empty()) && "Invalid instruction");
454 
455  if(!in.empty()) {
456  // if the current instruction has rvalue, which is two
457  // length expression
458 
459  // determine the operator and check if it is valid.
460  // If the operator is invalid then op will have value
461  // '\0', by which it was initialised
462  if(in.length() == 1) {
463  switch(in[0]) {
464  case '+':
465  case '-':
466  case '*':
467  case '/': op = in[0];
468  }
469  }
470  assert(op != '\0' && "Invalid operand");
471 
472  // read the second operand from the instruction buffer
473  in.clear(), ss >> in;
474  assert(!in.empty() && "Second operand not specified");
475 
476  // store right operand of rvalue of current instruction
477  try {
478  constVal = std::stoi(in.c_str());
479  rightOp = {true, Constants.insert(constVal).first};
480  } catch(std::invalid_argument) {
481  rightOp = {false, Variables.insert(in).first};
482  }
483  } else {
484  // if rvalue is not two length expression, a default value
485  // value for right operand for consistency
486  rightOp = {false, -1};
487  }
488  }
489 
490  // push the current instruction in `Instructions` vector
491  // at index `instCnt`. For now `reachable` and `cfgIndex`
492  // fields are set to default value of false and -1. These
493  // would be set to actual values later.
494  Instructions.emplace_back(InstructionTy({lValue, {op, leftOp, rightOp}, false, -1, {}}));
495  jumps.emplace_back(std::vector<std::string>());
496  instCnt++;
497  }
498  }
499 
500  // updating `jumps` and `Program::Instructions` for END instruction
501  Instructions.emplace_back(dummy);
502  jumps.emplace_back(std::vector<std::string>());
503  instCnt++;
504 
505  // now process the jumps as given by the labels. Reachability
506  // is determined by performing breadth first search (BFS) from
507  // index 0 in `Instructions` (the START instruction).
508  // Instructions for which `jumps` list is empty has by default
509  // jump to the instruction corresponding to the next index.
510  std::queue<int> q;
511  q.push(0), Instructions[0].reachable = true;
512 
513  while(not q.empty()) {
514  int cur = q.front();
515  q.pop();
516 
517  if(not jumps[cur].empty()) {
518  // process the jumps
519  for(auto el : jumps[cur]) {
520  auto it = labels.find(el);
521  assert(it != labels.end() && "Undefined label found");
522 
523  // if the label points to index `instCnt`, it means
524  // the current instruction being processed is the dummy
525  // END instruction and this has to be ignored
526  if(it->second == instCnt) continue;
527 
528  Instructions[it->second].predecessors.insert(cur);
529  if(not Instructions[it->second].reachable) {
530  Instructions[it->second].reachable = true;
531  q.push(it->second);
532  }
533  }
534  } else {
535  // if the index of the next instruction is `instCnt`, it
536  // means the current instruction being processed is the
537  // dummy END instruction and this has to be ignored
538  if(cur + 1 == instCnt) continue;
539 
540  // the successor of current instruction is the next
541  // instruction index
542  Instructions[cur + 1].predecessors.insert(cur);
543  if(not Instructions[cur + 1].reachable) {
544  Instructions[cur + 1].reachable = true;
545  q.push(cur + 1);
546  }
547  }
548  }
549 
550  fin.close();
551 }

602 void Program::createCFG() {
603  // holds the size of the control flow graph
604  int cfgSize = 0;
605 
606  // first set `cfgIndex` for each instruction
607  for(auto &el : Instructions) {
608  // there are no node in CFG corresponding to unreachable
609  // instructions, so we skip them
610  if(not el.reachable) continue;
611 
612  if(el.predecessors.size() > 1) {
613  // if the current instruction is a confluence point.
614  // Note that this condition is not `!= 1`, because it
615  // must also be false for the START instruction (for which
616  // `== 0` is true and hence `!= 1` would then be true)
617 
618  // there would be a node for the instruction as well as
619  // for the confluence point. The confluence point gets
620  // index `cfgSize` and the instruction gets index
621  // `cfgSize + 1`. Only 'cfgSize + 1' is stored, and
622  // the index of confluence point in CFG can be inferred
623  // with its help
624  el.cfgIndex = cfgSize + 1;
625 
626  // update `cfgSize`
627  cfgSize += 2;
628  } else {
629  // there would only be node for the instruction which
630  // gets index `cfgSize` and `cfgSize` is incremented
631  el.cfgIndex = cfgSize++;
632  }
633  }
634 
635  // now create control flow graph, using assigned `cfgIndex`
636  // for each instructions
637  CFG.resize(cfgSize);
638 
639  // for the 0th and (cfgSize - 1) index in `CFG`, `instructionIndex` is
640  // -1 to mark that they doesn't correspond to a valid instruction
641  CFG[0].instructionIndex = -1;
642  CFG[cfgSize - 1].instructionIndex = -1;
643 
644  for(int i = 1; i < Instructions.size(); i++) {
645  InstructionTy &I = Instructions[i];
646 
647  // ignore unreachable instructions
648  if(not I.reachable) continue;
649 
650  int idx = I.cfgIndex;
651  int sz = I.predecessors.size();
652 
653  if(sz > 1) {
654  // if the instruction has more than one predecessors
655 
656  // node for the confluence point
657  int confIdx = idx - 1;
658  CFG[confIdx].instructionIndex = -1;
659  for(auto el : I.predecessors) {
660  int predIdx = Instructions[el].cfgIndex;
661  CFG[confIdx].predecessors.push_back(predIdx);
662  }
663 
664  // node for the instruction
665  CFG[idx].predecessors.push_back(confIdx);
666  CFG[idx].instructionIndex = i;
667  } else {
668  // if the instruction is the first instruction in
669  // the program - its predecessor would be node 0
670  // in the CFG
671  int predIdx = 0;
672  if(sz != 0)
673  predIdx = Instructions[*(I.predecessors.begin())].cfgIndex;
674 
675  CFG[idx].predecessors.push_back(predIdx);
676  CFG[idx].instructionIndex = i;
677  }
678  }
679 }
