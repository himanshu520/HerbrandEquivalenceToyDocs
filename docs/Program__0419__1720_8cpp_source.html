341 void Program::parse(std::string fname) {
342  std::ifstream fin(fname);
343  assert(fin && "Error opening file");
344 
345  // keeps count of the current instructions number starting from 0
346  int instCnt = 0;
347 
348  // holds a program line
349  std::string buf;
350 
351  // holds jump labels for instructions corresponding to their indexes
352  std::vector<std::vector<std::string>> jumps;
353 
354  // holds the instruction indexes to which the labels refer to
355  std::map<std::string, int> labels;
356 
357  // read the program, processing each instruction type and checking
358  // the validity. The jumps can not be resolved at this stage (before
359  // the whole program is processed), as labels can be used before
360  // definitions
361  while(getline(fin, buf)) {
362  // ignores empty lines in the program text
363  if(buf.empty()) continue;
364 
365  // buffer holding current instruction
366  std::stringstream ss(buf);
367 
368  // holds the tokens in the instruction while they are processed
369  // in succession
370  std::string in;
371 
372  // read the first token in the instruction
373  ss >> in;
374 
375  if(in == "GOTO") {
376  // current instruction is a jump instruction
377  assert(instCnt != 0 && "GOTO occurs before first valid instruction");
378  while(ss >> in)
379  jumps[instCnt - 1].push_back(in);
380 
381  } else if(in == "LABEL") {
382  // current instruction defines a label
383  while(ss >> in) {
384  assert(labels.find(in) == labels.end() && "Duplicate label found");
385  labels[in] = instCnt;
386  }
387 
388  } else {
389  // current instruction is an assignment instruction
390 
391  int constVal;
392  ValueTy lValue, leftOp, rightOp;
393  char op = '\0';
394 
395  try {
396  // ideally lvalue should be a variable (a string that does
397  // not starts with a digit). So this must throw an exception
398  // and we should reach catch block.
399  constVal = std::stoi(in.c_str());
400 
401  assert(false && "LValue is not a variable");
402  } catch(std::invalid_argument) {
403  // storing lvalue of current instruction
404  lValue = {false, Variables.insert(in).first};
405 
406  // read and ignore the '=' in the instruction buffer
407  in.clear(), ss >> in;
408  assert(in == "=" && "No RValue specified");
409 
410  // read the first operand from the instruction buffer
411  in.clear(), ss >> in;
412  assert(!in.empty() && "No RValue specified");
413 
414  // store left operand of rvalue of current instruction
415  try {
416  constVal = std::stoi(in.c_str());
417  leftOp = {true, Constants.insert(constVal).first};
418  } catch(std::invalid_argument) {
419  if(in == "*") op = '#', leftOp = {false, -1};
420  else leftOp = {false, Variables.insert(in).first};
421  }
422 
423  // read the next token in the instruction buffer (which should
424  // be the operator if any)
425  in.clear(), ss >> in;
426 
427  // if the current instruction is non-deterministic assignment
428  // then it can not have a second operand. Note that if the
429  // rvalue is non-deterministic only then `op` has value '#'.
430  assert((op != '#' || in.empty()) && "Invalid instruction");
431 
432  if(!in.empty()) {
433  // if the current instruction has rvalue, which is two
434  // length expression
435 
436  // determine the operator and check if it is valid.
437  // If the operator is invalid then op will have value
438  // '\0', by which it was initialised
439  if(in.length() == 1) {
440  switch(in[0]) {
441  case '+':
442  case '-':
443  case '*':
444  case '/': op = in[0];
445  }
446  }
447  assert(op != '\0' && "Invalid operand");
448 
449  // read the second operand from the instruction buffer
450  in.clear(), ss >> in;
451  assert(!in.empty() && "Second operand not specified");
452 
453  // store right operand of rvalue of current instruction
454  try {
455  constVal = std::stoi(in.c_str());
456  rightOp = {true, Constants.insert(constVal).first};
457  } catch(std::invalid_argument) {
458  rightOp = {false, Variables.insert(in).first};
459  }
460  } else {
461  // if rvalue is not two length expression, a default value
462  // for consistency
463  rightOp = {false, -1};
464  }
465  }
466 
467  // push the current instruction in `Instructions` vector
468  // at index `instCnt`. For now `reachable` and `cfgIndex`
469  // fields are set to default value of false and -1. These
470  // would be set to actual values later.
471  Instructions.emplace_back(InstructionTy({lValue, {op, leftOp, rightOp}, false, -1, {}}));
472  instCnt++;
473  }
474  }
475 
476  // now process the jumps as given by the labels. Reachability
477  // is determined by performing breadth first search (BFS) from
478  // index 0 in `Instructions` (the starting instruction).
479  // Instructions for which `jumps` list is empty has by default
480  // jump to the instruction corresponding to the next index.
481  std::queue<int> q;
482  q.push(0), Instructions[0].reachable = true;
483 
484  while(not q.empty()) {
485  int cur = q.front();
486  q.pop();
487 
488  if(not jumps[cur].empty()) {
489  for(auto el : jumps[cur]) {
490  auto it = labels.find(el);
491  assert(it != labels.end() && "Undefined label found");
492 
493  // if the labels points to index `instCnt`, it means
494  // the end of the program and hence this is to be
495  // ignored
496  if(it->second == instCnt) continue;
497 
498  Instructions[it->second].predecessors.insert(cur);
499  if(not Instructions[it->second].reachable) {
500  Instructions[it->second].reachable = true;
501  q.push(it->second);
502  }
503  }
504  } else {
505  // if the index of the next instruction is `instCnt`, it
506  // means the end of the program and hence this is to be
507  // ignored
508  if(cur + 1 == instCnt) continue;
509 
510  Instructions[cur + 1].predecessors.insert(cur);
511  if(not Instructions[cur + 1].reachable) {
512  Instructions[cur + 1].reachable = true;
513  q.push(cur + 1);
514  }
515  }
516  }
517 
518  fin.close();
519 }

564 void Program::createCFG() {
565  // holds the current control flow graph index. cfgIdx
566  // starts with 1, because first node is fixed as the
567  // START point of the program with index 0
568  int cfgIdx = 1;
569 
570  // first set `cfgIndex` for each instruction
571  for(auto &el : Instructions) {
572  // there are no node in CFG corresponding to unreachable
573  // instructions, so we skip them
574  if(not el.reachable) continue;
575 
576  if(el.predecessors.size() > 1) {
577  // if the current instruction is a confluence point.
578  // Note that this condition is not `!= 1`, because it
579  // must also be false for the first instruction (for which
580  // `== 0` is true and hence `!= 1` would then be true)
581 
582  // there would be a node for the instruction as well as
583  // for the confluence point. The confluence point gets
584  // index `cfgIdx` and the instruction gets index
585  // `cfgIdx + 1`
586  el.cfgIndex = cfgIdx + 1;
587 
588  // update `cfgIdx` to next available index
589  cfgIdx += 2;
590  } else {
591  // there would only be node for the instruction which
592  // gets index `cfgIdx` and `cfgIdx` is incremented
593  el.cfgIndex = cfgIdx++;
594  }
595  }
596 
597  // now create control flow graph, using assigned `cfgIndex`
598  // for the instructions
599  CFG.resize(cfgIdx);
600 
601  // for the 0th index in `CFG`, `instructionIndex` is -1 to mark
602  // that it doesn't correspond to a valid instruction
603  CFG[0].instructionIndex = -1;
604 
605  for(int i = 0; i < Instructions.size(); i++) {
606  InstructionTy &I = Instructions[i];
607 
608  // ignore unreachable instructions
609  if(not I.reachable) continue;
610 
611  int idx = I.cfgIndex;
612  int sz = I.predecessors.size();
613 
614  if(sz > 1) {
615  // if the instruction has more than one predecessors
616 
617  // node for the confluence point
618  int confIdx = idx - 1;
619  CFG[confIdx].instructionIndex = -1;
620  for(auto el : I.predecessors) {
621  int predIdx = Instructions[el].cfgIndex;
622  CFG[confIdx].predecessors.push_back(predIdx);
623  }
624 
625  // node for the instruction
626  CFG[idx].predecessors.push_back(confIdx);
627  CFG[idx].instructionIndex = i;
628  } else {
629  // if the instruction is the first instruction in
630  // the program - its predecessor would be node 0
631  // in the CFG
632  int predIdx = 0;
633  if(sz != 0)
634  predIdx = Instructions[*(I.predecessors.begin())].cfgIndex;
635 
636  CFG[idx].predecessors.push_back(predIdx);
637  CFG[idx].instructionIndex = i;
638  }
639  }
640 }
